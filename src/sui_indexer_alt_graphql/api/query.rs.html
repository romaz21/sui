80    /// Fetch a `Node` by its globally unique `ID`. Returns `null` if the node cannot be found (e.g., the underlying data was pruned or never existed).
81    async fn node(&self, ctx: &Context<'_>, id: Id) -> Result<Option<Node>, RpcError> {
82        let scope = self.scope(ctx)?;
83        Ok(match id {
84            Id::Address(a) => Some(Node::Address(Box::new(Address::with_address(scope, a)))),
85
86            Id::Checkpoint(s) => Checkpoint::with_sequence_number(scope, Some(s))
87                .map(Box::new)
88                .map(Node::Checkpoint),
89
90            Id::DynamicFieldByAddress(a) => {
91                let object = Object::with_address(scope, a);
92                DynamicField::from_object(&object, ctx)
93                    .await?
94                    .map(Box::new)
95                    .map(Node::DynamicField)
96            }
97
98            Id::DynamicFieldByRef(a, v, d) => {
99                let object = Object::with_ref(&scope, a, v, d);
100                DynamicField::from_object(&object, ctx)
101                    .await?
102                    .map(Box::new)
103                    .map(Node::DynamicField)
104            }
105
106            Id::Epoch(e) => Some(Node::Epoch(Box::new(Epoch::with_id(scope, e)))),
107
108            Id::MoveObjectByAddress(a) => {
109                let object = Object::with_address(scope, a);
110                MoveObject::from_object(&object, ctx)
111                    .await?
112                    .map(Box::new)
113                    .map(Node::MoveObject)
114            }
115
116            Id::MoveObjectByRef(a, v, d) => {
117                let object = Object::with_ref(&scope, a, v, d);
118                MoveObject::from_object(&object, ctx)
119                    .await?
120                    .map(Box::new)
121                    .map(Node::MoveObject)
122            }
123
124            Id::MovePackage(a) => Some(Node::MovePackage(Box::new(MovePackage::with_address(
125                scope, a,
126            )))),
127
128            Id::ObjectByAddress(a) => Some(Node::Object(Box::new(Object::with_address(scope, a)))),
129
130            Id::ObjectByRef(a, v, d) => {
131                Some(Node::Object(Box::new(Object::with_ref(&scope, a, v, d))))
132            }
133
134            Id::Transaction(d) => Some(Node::Transaction(Box::new(Transaction::with_digest(
135                scope, d,
136            )))),
137        })
138    }

140    /// Look-up an account by its SuiAddress.
141    ///
142    /// If `rootVersion` is specified, nested dynamic field accesses will be fetched at or before this version. This can be used to fetch a child or descendant object bounded by its root object's version, when its immediate parent is wrapped, or a value in a dynamic object field. For any wrapped or child (object-owned) object, its root object can be defined recursively as:
143    ///
144    /// - The root object of the object it is wrapped in, if it is wrapped.
145    /// - The root object of its owner, if it is owned by another object.
146    /// - The object itself, if it is not object-owned or wrapped.
147    ///
148    /// Specifying a `rootVersion` disables nested queries for paginating owned objects or dynamic fields (these queries are only supported at checkpoint boundaries).
149    ///
150    /// If `atCheckpoint` is specified, the address will be fetched at the latest version as of this checkpoint. This will fail if the provided checkpoint is after the RPC's latest checkpoint.
151    ///
152    /// If none of the above are specified, the address is fetched at the checkpoint being viewed.
153    ///
154    /// If the address is fetched by name and the name does not resolve to an address (e.g. the name does not exist or has expired), `null` is returned.
155    async fn address(
156        &self,
157        ctx: &Context<'_>,
158        address: Option<SuiAddress>,
159        name: Option<Domain>,
160        root_version: Option<UInt53>,
161        at_checkpoint: Option<UInt53>,
162    ) -> Result<Option<Address>, RpcError<address::Error>> {
163        Address::by_key(
164            ctx,
165            self.scope(ctx)?,
166            AddressKey {
167                address,
168                name,
169                root_version,
170                at_checkpoint,
171            },
172        )
173        .await
174    }

197    /// Paginate checkpoints in the network, optionally bounded to checkpoints in the given epoch.
198    async fn checkpoints(
199        &self,
200        ctx: &Context<'_>,
201        first: Option<u64>,
202        after: Option<CCheckpoint>,
203        last: Option<u64>,
204        before: Option<CCheckpoint>,
205        filter: Option<CheckpointFilter>,
206    ) -> Result<Option<Connection<String, Checkpoint>>, RpcError> {
207        let scope = self.scope(ctx)?;
208        let pagination: &PaginationConfig = ctx.data()?;
209        let limits = pagination.limits("Query", "checkpoints");
210        let page = Page::from_params(limits, first, after, last, before)?;
211
212        let filter = filter.unwrap_or_default();
213        Checkpoint::paginate(ctx, scope, page, filter)
214            .await
215            .map(Some)
216    }

668    /// Simulate a transaction to preview its effects without executing it on chain.
669    ///
670    /// Accepts a JSON transaction matching the [Sui gRPC API schema](https://docs.sui.io/references/fullnode-protocol#sui-rpc-v2-Transaction).
671    /// The JSON format allows for partial transaction specification where certain fields can be automatically resolved by the server.
672    ///
673    /// Alternatively, for already serialized transactions, you can pass BCS-encoded data:
674    /// `{"bcs": {"value": "<base64>"}}`
675    ///
676    /// Unlike `executeTransaction`, this does not require signatures since the transaction is not committed to the blockchain. This allows for previewing transaction effects, estimating gas costs, and testing transaction logic without spending gas or requiring valid signatures.
677    ///
678    /// - `checksEnabled`: If true, enables transaction validation checks during simulation. Defaults to true.
679    /// - `doGasSelection`: If true, enables automatic gas coin selection and budget estimation. Defaults to false.
680    async fn simulate_transaction(
681        &self,
682        ctx: &Context<'_>,
683        transaction: Json,
684        checks_enabled: Option<bool>,
685        do_gas_selection: Option<bool>,
686    ) -> Result<SimulationResult, RpcError<TransactionInputError>> {
687        let fullnode_client: &FullnodeClient = ctx.data()?;
688
689        // Convert Json to serde_json::Value and parse as proto::Transaction
690        let json_value: serde_json::Value = transaction
691            .try_into()
692            .map_err(|err| bad_user_input(TransactionInputError::InvalidTransactionJson(err)))?;
693        let proto_tx: proto::Transaction = serde_json::from_value(json_value)
694            .map_err(|err| bad_user_input(TransactionInputError::InvalidTransactionJson(err)))?;
695
696        // Simulate transaction using proto
697        match fullnode_client
698            .simulate_transaction(
699                proto_tx,
700                checks_enabled.unwrap_or(true),
701                do_gas_selection.unwrap_or(false),
702            )
703            .await
704        {
705            Ok(response) => {
706                let scope = self.scope(ctx)?;
707                let tx_data = response
708                    .transaction
709                    .as_ref()
710                    .and_then(|executed_tx| executed_tx.transaction.as_ref())
711                    .and_then(|tx| tx.bcs.as_ref())
712                    .ok_or_else(|| anyhow!("Missing transaction or BCS in simulation response"))?
713                    .deserialize()
714                    .context("Failed to deserialize transaction from response")?;
715
716                SimulationResult::from_simulation_response(scope, response, tx_data).map_err(upcast)
717            }
718            Err(GrpcExecutionError(status)) => Ok(SimulationResult {
719                effects: None,
720                outputs: None,
721                error: Some(status.to_string()),
722            }),
723            Err(other_error) => Err(anyhow!(other_error)
724                .context("Failed to simulate transaction")
725                .into()),
726        }
727    }
